Typography utility classes

/* Most reached-for combinations */ /* Display title */ .txt-display-xl { font-family: var(--font-display); font-weight: 900; font-size: 64px; line-height: 0.88; letter-spacing: -0.025em; text-transform: uppercase; } /* Eyebrow label */ .txt-eyebrow { font-family: var(--font-mono); font-size: 10.5px; letter-spacing: 0.22em; text-transform: uppercase; color: var(--text-dim); } /* Data label */ .txt-data { font-family: var(--font-mono); font-size: 11px; letter-spacing: 0.08em; font-feature-settings: "tnum"; }

Noise overlay recipe

/* Applied globally to body::before, pointer-events:none */ body::before { content: ''; position: fixed; inset: 0; pointer-events: none; opacity: 0.04; z-index: 1000; mix-blend-mode: overlay; background-image: url("data:image/svg+xml,..."); /* inline SVG turbulence filter */ }

Motion

/* Fade-up entrance — most common */ @keyframes fadeUp { from { opacity: 0; transform: translateY(12px); } to { opacity: 1; transform: translateY(0); } } /* Staggered children · reveal rhythm */ .fade-up { animation: fadeUp 0.6s cubic-bezier(.2,.8,.2,1) forwards; } .delay-1 { animation-delay: 0.05s; } .delay-2 { animation-delay: 0.10s; } .delay-3 { animation-delay: 0.15s; } /* ... step size 50ms */
